Proste programy i algorytmy w Python

Proste programy w Pythonie i algorytmy które wykorzystywałem w różnych projektach.

            
              
              
              # Prosty kalkulator w Pythonie

              # Funkcja dodawania
              def dodaj(a, b):
                  return a + b

              # Funkcja odejmowania
              def odejmij(a, b):
                  return a - b

              # Funkcja mnożenia
              def pomnóż(a, b):
                  return a * b

              # Funkcja dzielenia
              def podziel(a, b):
                  if b != 0:
                      return a / b
                  else:
                      return "Nie można dzielić przez zero!"

              # Pobranie danych od użytkownika
              num1 = float(input("Podaj pierwszą liczbę: "))
              operator = input("Podaj operatora (+, -, *, /): ")
              num2 = float(input("Podaj drugą liczbę: "))

              # Wybór operacji i wyświetlenie wyniku
              if operator == '+':
                  print("Wynik dodawania:", dodaj(num1, num2))
              elif operator == '-':
                  print("Wynik odejmowania:", odejmij(num1, num2))
              elif operator == '*':
                  print("Wynik mnożenia:", pomnóż(num1, num2))
              elif operator == '/':
                  print("Wynik dzielenia:", podziel(num1, num2))
              else:
                  print("Nieprawidłowy operator!")
